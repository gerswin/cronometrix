<!-- Backend endpoints (delivered by 07-01-PLAN.md) the frontend consumes. -->

POST /api/v1/enrollments (multipart, require_admin)
  FormData fields: employee_id (UUID string), captured_via ("device" | "webcam" | "upload"),
                   source_device_id (UUID string, required only when captured_via=="device"),
                   photo (Blob, JPEG, ≤2MB), face_quality_score (optional JSON string)
  Returns 202 + JSON: { enrollment_id: string, face_id: string, device_pushes: EnrollmentDevicePush[] }

GET /api/v1/enrollments/:id (require_admin)
  Returns 200 + JSON: Enrollment shape (see types below)

POST /api/v1/enrollments/:id/devices/:device_id/retry (require_admin, no body)
  Returns 202 + JSON: { ok: true }

POST /api/v1/enrollments/capture-from-device (require_admin)
  Body: { device_id: string, employee_id: string }
  Returns 202 + JSON: { capture_id: string, status: "capturing" }

GET /api/v1/enrollments/captures/:capture_id (require_admin)
  Returns 200 + JSON: { capture_id: string, status: "capturing"|"captured"|"timeout"|"error",
                        photo_path?: string, error_message?: string }

         video: HTMLVideoElement,
         canvas: HTMLCanvasElement,
         faceapi: typeof import('@vladmandic/face-api'),
       ): Promise<FrameAnalysis> {

         const luminanceOk = luminance >= 80 && luminance <= 200
         return {

           luminance,
           width: det?.box.width ?? 0,
