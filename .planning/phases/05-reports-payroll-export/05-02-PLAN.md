**W-5 fix (leave-day counting):** The primary daily_records JOIN only sees days where the engine attached a leave overlay (dr.leave_id NOT NULL). Full-week leaves on dates with zero captures (e.g., vacation week with no biometric events) would be invisible to the JOIN — so días_vacación / IVSS / permiso / no-remunerado would under-count. The service runs a SECOND aggregation directly against `leaves` to count overlap days per (employee, leave_type) within [from..to], then MERGES the results into the per-employee accumulator. The two sources are reconciled: leave-day counts come exclusively from the leaves-table aggregation; the daily_records branch only handles money math (vacation paid-full pay, medical zero pay, etc.) and never increments leave-day counters. This guarantees a vacation week with no captures shows días_vacación = 5 (Mon-Fri) and días_ausentes = 0 (the days are accounted for via the leave overlay in días_ausentes computation).

**W-6 fix (shift_type source):** Night-premium gating reads `daily_records.shift_type` (the per-day actual shift recorded by the engine in Phase 3, migration 007 line 12), NOT `departments.shift_type` (the policy/default). This captures shift overrides if they exist (engine output is authoritative for what actually happened on each day). A code comment in service.rs makes this choice explicit, and an integration test asserts the behavior.

```rust
pub fn work_pay_cents(work_minutes: i64, base_salary_cents: i64, ordinary_daily_minutes: i64) -> i64 {
    if ordinary_daily_minutes <= 0 { return 0; }
    work_minutes
        .checked_mul(base_salary_cents)
        .map(|p| p / ordinary_daily_minutes)
        .unwrap_or(0)
}

pub fn ot_pay_cents(ot_minutes: i64, base_salary_cents: i64, ordinary_daily_minutes: i64) -> i64 {
    if ordinary_daily_minutes <= 0 { return 0; }
    ot_minutes
        .checked_mul(base_salary_cents)
        .and_then(|p| p.checked_mul(150))
        .map(|p| p / (100 * ordinary_daily_minutes))
        .unwrap_or(0)
}

pub fn night_premium_cents(work_minutes: i64, base_salary_cents: i64, ordinary_daily_minutes: i64) -> i64 {
    if ordinary_daily_minutes <= 0 { return 0; }
    work_minutes
        .checked_mul(base_salary_cents)
        .and_then(|p| p.checked_mul(30))
        .map(|p| p / (100 * ordinary_daily_minutes))
        .unwrap_or(0)
}

pub fn rest_day_surcharge_cents(work_minutes: i64, base_salary_cents: i64, ordinary_daily_minutes: i64) -> i64 {
    if ordinary_daily_minutes <= 0 { return 0; }
    work_minutes
        .checked_mul(base_salary_cents)
        .and_then(|p| p.checked_mul(50))
        .map(|p| p / (100 * ordinary_daily_minutes))
        .unwrap_or(0)
}

pub fn late_deduction_cents(late_minutes: i64, base_salary_cents: i64, ordinary_daily_minutes: i64) -> i64 {
    if ordinary_daily_minutes <= 0 { return 0; }
    late_minutes
        .checked_mul(base_salary_cents)
        .map(|p| p / ordinary_daily_minutes)
        .unwrap_or(0)
}

pub fn total_a_pagar_cents(work_pay: i64, ot_pay: i64, night_premium: i64, rest_day_surcharge: i64, late_deduction: i64) -> i64 {
    work_pay
        .saturating_add(ot_pay)
        .saturating_add(night_premium)
        .saturating_add(rest_day_surcharge)
        .saturating_sub(late_deduction)
}
```

```rust
use chrono::{Datelike, Duration, NaiveDate};

#[derive(Debug, Clone, Copy, PartialEq)]
pub enum PeriodPreset {
    Weekly,
    BiweeklyFirst,
    BiweeklySecond,
    Monthly,
    Custom(NaiveDate, NaiveDate),
}

pub fn resolve_period(preset: PeriodPreset, ref_date: NaiveDate) -> (NaiveDate, NaiveDate) {
    match preset {
        PeriodPreset::Weekly => {
            let dow = ref_date.weekday().num_days_from_monday() as i64;
            let mon = ref_date - Duration::days(dow);
            let sun = mon + Duration::days(6);
            (mon, sun)
        }
        PeriodPreset::BiweeklyFirst => {
            let first = NaiveDate::from_ymd_opt(ref_date.year(), ref_date.month(), 1).unwrap();
            let fifteenth = NaiveDate::from_ymd_opt(ref_date.year(), ref_date.month(), 15).unwrap();
            (first, fifteenth)
        }
        PeriodPreset::BiweeklySecond => {
            let sixteenth = NaiveDate::from_ymd_opt(ref_date.year(), ref_date.month(), 16).unwrap();
            let eom = last_day_of_month(ref_date.year(), ref_date.month());
            (sixteenth, eom)
        }
        PeriodPreset::Monthly => {
            let first = NaiveDate::from_ymd_opt(ref_date.year(), ref_date.month(), 1).unwrap();
            let eom = last_day_of_month(ref_date.year(), ref_date.month());
            (first, eom)
        }
        PeriodPreset::Custom(from, to) => (from, to),
    }
}

fn last_day_of_month(year: i32, month: u32) -> NaiveDate {
    let (next_year, next_month) = if month == 12 { (year + 1, 1) } else { (year, month + 1) };
    NaiveDate::from_ymd_opt(next_year, next_month, 1).unwrap() - Duration::days(1)
}

/// Parse a period_type string + (from_date, to_date) into a PeriodPreset.
/// Returns AppError::Validation if the string is unknown.
pub fn parse_period(period_type: &str, from_date: &str, to_date: &str) -> Result<PeriodPreset, crate::errors::AppError> {
    use crate::errors::AppError;
    let from = NaiveDate::parse_from_str(from_date, "%Y-%m-%d")
        .map_err(|_| AppError::Validation { code: "VALIDATION_ERROR", message: "from_date must be YYYY-MM-DD".to_string() })?;
    let to = NaiveDate::parse_from_str(to_date, "%Y-%m-%d")
        .map_err(|_| AppError::Validation { code: "VALIDATION_ERROR", message: "to_date must be YYYY-MM-DD".to_string() })?;
    match period_type {
        "weekly" => Ok(PeriodPreset::Weekly),
        "biweekly_first" => Ok(PeriodPreset::BiweeklyFirst),
        "biweekly_second" => Ok(PeriodPreset::BiweeklySecond),
        "monthly" => Ok(PeriodPreset::Monthly),
        "custom" => Ok(PeriodPreset::Custom(from, to)),
        other => Err(AppError::Validation { code: "VALIDATION_ERROR", message: format!("Unknown period_type: {}", other) }),
    }
}
```

```rust
pub async fn compute_report(
    state: &AppState,
    actor_id: &str,
    params: &ReportParamsRequest,
    format: &str,
) -> Result<ReportPayload, AppError> {
    // 1. Parse period_type → PeriodPreset → (from, to). Honor params.from_date/to_date for "custom".
    let preset = periods::parse_period(&params.period_type, &params.from_date, &params.to_date)?;

    // ref_date for non-custom presets: use from_date as the anchor
    let ref_date = NaiveDate::parse_from_str(&params.from_date, "%Y-%m-%d")
        .map_err(|_| AppError::Validation { code: "VALIDATION_ERROR", message: "from_date must be YYYY-MM-DD".to_string() })?;
    let (from, to) = periods::resolve_period(preset, ref_date);

    // DoS guard: max 366-day range. AppError::Validation maps to HTTP 422 per errors.rs:88-89.
    if (to - from).num_days() > 366 {
        return Err(AppError::Validation { code: "VALIDATION_ERROR", message: "Period range cannot exceed 366 days".to_string() });
    }
    if from > to {
        return Err(AppError::Validation { code: "VALIDATION_ERROR", message: "from_date must be ≤ to_date".to_string() });
    }

    let conn = state.db.connect().map_err(|e| AppError::Internal(e.into()))?;

    // 2. Fetch tenant_info for branding header.
    let tenant = conn
        .query("SELECT client_name, client_rif FROM tenant_info WHERE id = 1", ())
        .await.map_err(|e| AppError::Internal(e.into()))?
        .next().await.map_err(|e| AppError::Internal(e.into()))?
        .ok_or_else(|| AppError::Internal(anyhow::anyhow!("tenant_info row missing")))?;

    let client_rif: String = tenant.get(1).map_err(|e| AppError::Internal(e.into()))?;

        client_name,
        client_rif,
        from_date: from.to_string(),
        to_date: to.to_string(),
        generated_at_iso: epoch_to_iso(chrono::Utc::now().timestamp()),
    };

    // 3. Build dynamic SQL with parameterized predicates for the daily_records JOIN.
    let mut predicates: Vec<String> = vec![
        "dr.anchor_date BETWEEN ?1 AND ?2".to_string(),

            Some("vacation") => {
                entry.leave_dates.insert(anchor_date);
                // Vacation paid full at ordinary daily salary (only when overlay attached to a daily_record).
                let work_pay = money::work_pay_cents(ordinary_daily_minutes, base_salary_cents, ordinary_daily_minutes);
                entry.agg.work_pay_cents = entry.agg.work_pay_cents.saturating_add(work_pay);
                entry.agg.total_a_pagar_cents = entry.agg.total_a_pagar_cents.saturating_add(work_pay);
            }
            Some("unpaid") => {
                entry.leave_dates.insert(anchor_date);
            }
            Some("manual") => {
                entry.leave_dates.insert(anchor_date);
            }
            _ => {
                // Standard work day money math
                let work_pay = money::work_pay_cents(effective_work_min, base_salary_cents, ordinary_daily_minutes);
                let ot_pay = money::ot_pay_cents(overtime_minutes, base_salary_cents, ordinary_daily_minutes);
                // W-6: night premium gates on dr.shift_type (per-day actual shift), NOT departments.shift_type.
                let night = if day_shift_type == "night" {
                    money::night_premium_cents(effective_work_min, base_salary_cents, ordinary_daily_minutes)
                } else { 0 };
                let rest = if is_rest_day_worked == 1 {
                    money::rest_day_surcharge_cents(effective_work_min, base_salary_cents, ordinary_daily_minutes)
                } else { 0 };
                let late = money::late_deduction_cents(late_minutes, base_salary_cents, ordinary_daily_minutes);
                let total = money::total_a_pagar_cents(work_pay, ot_pay, night, rest, late);

                entry.agg.work_min = entry.agg.work_min.saturating_add(effective_work_min);
                entry.agg.ot_min = entry.agg.ot_min.saturating_add(overtime_minutes);
                entry.agg.late_min = entry.agg.late_min.saturating_add(late_minutes);
                entry.agg.work_pay_cents = entry.agg.work_pay_cents.saturating_add(work_pay);
                entry.agg.ot_pay_cents = entry.agg.ot_pay_cents.saturating_add(ot_pay);
                entry.agg.night_premium_cents = entry.agg.night_premium_cents.saturating_add(night);
                entry.agg.rest_day_surcharge_cents = entry.agg.rest_day_surcharge_cents.saturating_add(rest);
                entry.agg.late_deduction_cents = entry.agg.late_deduction_cents.saturating_add(late);
                entry.agg.total_a_pagar_cents = entry.agg.total_a_pagar_cents.saturating_add(total);
                if effective_work_min > 0 {
                    entry.agg.days_worked += 1;
                    entry.worked_dates.insert(anchor_date);
                }
            }
        }

        if let Some(s) = anomaly_codes_str_opt.filter(|s| !s.is_empty()) {
            for code in s.split(',') {
                entry.anomaly_codes_set.insert(code.to_string());
            }
        }
    }

    // 5. **W-5 fix — secondary leaves aggregation.**
    // The daily_records JOIN above only sees days where the engine attached a leave overlay.
    // Full-period leaves on dates with NO captures (e.g., vacation week with zero biometric events)
    // would be invisible — under-counting días_vacación / IVSS / permiso / no-remunerado.
    //
    // Run a second query directly against `leaves`, scoped to the same employee filter the JOIN
    // used. For each leave row, count overlap days with [from..to] and increment the counter
    // on the matching employee accumulator. Leave dates are also inserted into entry.leave_dates
    // so días_ausentes (computed below) correctly excludes them.
    //
    // Money treatment for leaves WITHOUT a daily_record overlay is documented as a known v1
    // limitation: only overlays attached to a daily_record produce vacation pay (the daily_records
    // branch above). Leave-only days produce only counter increments. Future work could synthesize
    // a vacation-pay row for leave-only days; out of scope for v1.

    let mut leave_predicates: Vec<String> = vec![
        "l.status = 'active'".to_string(),
        "l.deleted_at IS NULL".to_string(),
        "l.from_date <= ?1".to_string(),
        "l.to_date   >= ?2".to_string(),
        "e.id = l.employee_id".to_string(),
    ];
    let mut leave_values: Vec<libsql::Value> = vec![
        libsql::Value::Text(to.to_string()),
        libsql::Value::Text(from.to_string()),
    ];
    if !params.include_inactive.unwrap_or(false) {
        leave_predicates.push("e.status = 'active'".to_string());
    }
    if let Some(dept_ids) = &params.department_ids {
        if !dept_ids.is_empty() {
            let placeholders: Vec<String> = dept_ids.iter().enumerate()
                .map(|(i, _)| format!("?{}", leave_values.len() + 1 + i))
                .collect();
            leave_predicates.push(format!("e.department_id IN ({})", placeholders.join(",")));
            for id in dept_ids { leave_values.push(libsql::Value::Text(id.clone())); }
        }
    }
    if let Some(eid) = &params.employee_id {
        leave_predicates.push(format!("e.id = ?{}", leave_values.len() + 1));
        leave_values.push(libsql::Value::Text(eid.clone()));
    }
    let leave_where = format!("WHERE {}", leave_predicates.join(" AND "));

    let leave_sql = format!("
        SELECT l.employee_id, l.leave_type, l.from_date, l.to_date,
               e.employee_code, e.name, e.position,
               d.id AS dept_id, d.name AS dept_name,
               d.base_salary_cents, d.ordinary_daily_minutes,
               d.shift_type AS dept_shift_type
          FROM leaves l
          JOIN employees e   ON e.id = l.employee_id
          JOIN departments d ON d.id = e.department_id
          {leave_where}
    ");

    let mut leave_iter = conn.query(&leave_sql, libsql::params_from_iter(leave_values))
        .await.map_err(|e| AppError::Internal(e.into()))?;

    while let Some(lr) = leave_iter.next().await.map_err(|e| AppError::Internal(e.into()))? {
        let l_employee_id: String = lr.get(0).map_err(|e| AppError::Internal(e.into()))?;
        let l_type: String = lr.get(1).map_err(|e| AppError::Internal(e.into()))?;
        let l_from_str: String = lr.get(2).map_err(|e| AppError::Internal(e.into()))?;
        let l_to_str: String = lr.get(3).map_err(|e| AppError::Internal(e.into()))?;
        let l_cedula: String = lr.get(4).map_err(|e| AppError::Internal(e.into()))?;
        let l_nombre: String = lr.get(5).map_err(|e| AppError::Internal(e.into()))?;
        let l_cargo: String = lr.get(6).map_err(|e| AppError::Internal(e.into()))?;
        let l_dept_id: String = lr.get(7).map_err(|e| AppError::Internal(e.into()))?;
        let l_dept_name: String = lr.get(8).map_err(|e| AppError::Internal(e.into()))?;
        let l_base: i64 = lr.get(9).map_err(|e| AppError::Internal(e.into()))?;
        let l_ord: i64 = lr.get(10).map_err(|e| AppError::Internal(e.into()))?;
        let l_dept_shift: String = lr.get(11).map_err(|e| AppError::Internal(e.into()))?;

        let l_from = NaiveDate::parse_from_str(&l_from_str, "%Y-%m-%d").map_err(|e| AppError::Internal(anyhow::anyhow!(e)))?;
        let l_to   = NaiveDate::parse_from_str(&l_to_str,   "%Y-%m-%d").map_err(|e| AppError::Internal(anyhow::anyhow!(e)))?;
        let overlap_from = std::cmp::max(from, l_from);
        let overlap_to   = std::cmp::min(to, l_to);
        if overlap_from > overlap_to { continue; }

        // Ensure the employee has an accumulator entry even if they had no daily_records.
        let entry = acc.entry(l_employee_id.clone()).or_insert_with(|| {
            dept_order_seen.entry(l_dept_name.clone()).or_insert(l_dept_id.clone());
            AccRow {
                employee_id: l_employee_id.clone(),
                dept_id: l_dept_id.clone(),
                cedula: l_cedula.clone(),
                nombre: l_nombre.clone(),
                departamento: l_dept_name.clone(),
                cargo: l_cargo.clone(),
                shift_type: l_dept_shift.clone(),  // fallback: dept policy when no per-day shift seen
                base_salary_cents: l_base,
                ordinary_daily_minutes: l_ord,
                agg: Aggregates::default(),
                anomaly_codes_set: Default::default(),
                worked_dates: Default::default(),
                leave_dates: Default::default(),
            }
        });

        let mut d = overlap_from;
        while d <= overlap_to {
            entry.leave_dates.insert(d);
            match l_type.as_str() {
                "medical"  => entry.agg.days_ivss += 1,
                "vacation" => entry.agg.days_vacation += 1,
                "manual"   => entry.agg.days_permission += 1,
                "unpaid"   => entry.agg.days_unpaid += 1,
                _ => {}
            }
            d = d + Duration::days(1);
        }
    }

    // 6. Compute días_ausentes per employee — Mon-Fri in [from..=to] minus worked_dates minus leave_dates.
    let weekdays_in_period: Vec<NaiveDate> = (0..=(to - from).num_days())
        .map(|d| from + Duration::days(d))
        .filter(|d| d.weekday().num_days_from_monday() < 5)  // 0..=4 = Mon..Fri
        .collect();

    for entry in acc.values_mut() {
        let absent = weekdays_in_period.iter()
            .filter(|d| !entry.worked_dates.contains(d) && !entry.leave_dates.contains(d))
            .count() as i64;
        entry.agg.days_absent = absent;
    }

    // 7. Build EmployeeReportRow vec, departments_in_order, dept_subtotals, grand_total.
    let mut rows: Vec<EmployeeReportRow> = Vec::new();
    let mut dept_to_subtotal: BTreeMap<String, (String, Aggregates)> = BTreeMap::new();
    let mut grand = Aggregates::default();

    for (_, e) in acc {
        let codes_vec: Vec<String> = e.anomaly_codes_set.into_iter().collect();
        let count = codes_vec.len() as i64;

        // Accumulate dept subtotal
        let entry = dept_to_subtotal.entry(e.dept_id.clone()).or_insert_with(|| (e.departamento.clone(), Aggregates::default()));
        entry.1.work_min = entry.1.work_min.saturating_add(e.agg.work_min);
        entry.1.ot_min = entry.1.ot_min.saturating_add(e.agg.ot_min);
        entry.1.late_min = entry.1.late_min.saturating_add(e.agg.late_min);
        entry.1.days_worked = entry.1.days_worked.saturating_add(e.agg.days_worked);
        entry.1.days_absent = entry.1.days_absent.saturating_add(e.agg.days_absent);
        entry.1.work_pay_cents = entry.1.work_pay_cents.saturating_add(e.agg.work_pay_cents);
        entry.1.ot_pay_cents = entry.1.ot_pay_cents.saturating_add(e.agg.ot_pay_cents);
        entry.1.night_premium_cents = entry.1.night_premium_cents.saturating_add(e.agg.night_premium_cents);
        entry.1.rest_day_surcharge_cents = entry.1.rest_day_surcharge_cents.saturating_add(e.agg.rest_day_surcharge_cents);
        entry.1.late_deduction_cents = entry.1.late_deduction_cents.saturating_add(e.agg.late_deduction_cents);
        entry.1.total_a_pagar_cents = entry.1.total_a_pagar_cents.saturating_add(e.agg.total_a_pagar_cents);
        entry.1.days_ivss = entry.1.days_ivss.saturating_add(e.agg.days_ivss);
        entry.1.days_vacation = entry.1.days_vacation.saturating_add(e.agg.days_vacation);
        entry.1.days_permission = entry.1.days_permission.saturating_add(e.agg.days_permission);
        entry.1.days_unpaid = entry.1.days_unpaid.saturating_add(e.agg.days_unpaid);

        // Accumulate grand
        grand.work_min = grand.work_min.saturating_add(e.agg.work_min);
        grand.ot_min = grand.ot_min.saturating_add(e.agg.ot_min);
        grand.late_min = grand.late_min.saturating_add(e.agg.late_min);
        grand.days_worked = grand.days_worked.saturating_add(e.agg.days_worked);
        grand.days_absent = grand.days_absent.saturating_add(e.agg.days_absent);
        grand.work_pay_cents = grand.work_pay_cents.saturating_add(e.agg.work_pay_cents);
        grand.ot_pay_cents = grand.ot_pay_cents.saturating_add(e.agg.ot_pay_cents);
        grand.night_premium_cents = grand.night_premium_cents.saturating_add(e.agg.night_premium_cents);
        grand.rest_day_surcharge_cents = grand.rest_day_surcharge_cents.saturating_add(e.agg.rest_day_surcharge_cents);
        grand.late_deduction_cents = grand.late_deduction_cents.saturating_add(e.agg.late_deduction_cents);
        grand.total_a_pagar_cents = grand.total_a_pagar_cents.saturating_add(e.agg.total_a_pagar_cents);
        grand.days_ivss = grand.days_ivss.saturating_add(e.agg.days_ivss);
        grand.days_vacation = grand.days_vacation.saturating_add(e.agg.days_vacation);
        grand.days_permission = grand.days_permission.saturating_add(e.agg.days_permission);
        grand.days_unpaid = grand.days_unpaid.saturating_add(e.agg.days_unpaid);

        rows.push(EmployeeReportRow {
            employee_id: e.employee_id, dept_id: e.dept_id, cedula: e.cedula, nombre: e.nombre,
            departamento: e.departamento, cargo: e.cargo, shift_type: e.shift_type,
            aggregates: e.agg, anomaly_codes: codes_vec, anomaly_count: count,
        });
    }

    // departments_in_order — sorted by name (D-26)
    let mut dept_order: Vec<DeptSummary> = dept_to_subtotal.iter()
        .map(|(id, (name, _))| DeptSummary { id: id.clone(), name: name.clone() })
        .collect();
    dept_order.sort_by(|a, b| a.name.cmp(&b.name));

    let dept_subtotals: Vec<DeptSubtotal> = dept_order.iter().map(|d| {
        let (_, a) = dept_to_subtotal.get(&d.id).unwrap();
        DeptSubtotal { dept_id: d.id.clone(), dept_name: d.name.clone(), aggregates: a.clone() }
    }).collect();

    // 8. Audit insert AFTER success (Pitfall 7).
    write_export_audit(&conn, actor_id, params, format).await?;

    Ok(ReportPayload { header, rows, dept_subtotals, grand_total: grand, departments_in_order: dept_order })
}
```
