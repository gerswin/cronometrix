    5. backend/src/isapi/parser.rs — multer primary + line-scan fallback:

       ```rust
       use bytes::Bytes;
       use futures::Stream;
       use multer::Multipart;
       use std::pin::Pin;

       pub struct EventPair {
           pub xml: Bytes,
           pub jpeg: Option<Bytes>,
       }

       /// Parse a byte buffer into event pairs. Used for unit tests (fixture-driven).
       /// For live streams, use Multipart::new(bytes_stream, boundary).
       pub async fn parse_buffer(body: &[u8], boundary: &str) -> anyhow::Result<Vec<EventPair>> {
           let stream = futures::stream::once(async move { Ok::<_, std::io::Error>(Bytes::copy_from_slice(body)) });
           // T-2-19 mitigation (ASVS L1 V12): bound per-field + whole-stream sizes.
           let constraints = multer::Constraints::new()
               .size_limit(
                   multer::SizeLimit::new()
                       .per_field(10 * 1024 * 1024)        // 10 MB per part (JPEG ceiling)
                       .whole_stream(64 * 1024 * 1024),    // 64 MB per buffer parse
               );
           let mut mp = multer::Multipart::with_constraints(stream, boundary, constraints);
           let mut out = Vec::new();
           let mut pending_xml: Option<Bytes> = None;

           while let Some(field) = mp.next_field().await? {
               let ct = field.content_type().map(|m| m.to_string()).unwrap_or_default();
               let bytes = field.bytes().await?;
               if ct.starts_with("application/xml") || bytes.starts_with(b"<EventNotificationAlert") {
                   // Commit any previous pending XML photoless (Pitfall 2)
                   if let Some(prev) = pending_xml.take() {
                       out.push(EventPair { xml: prev, jpeg: None });
                   }
                   pending_xml = Some(bytes);
               } else if ct.starts_with("image/jpeg") || bytes.starts_with(b"\xFF\xD8\xFF") {
                   if let Some(prev) = pending_xml.take() {
                       out.push(EventPair { xml: prev, jpeg: Some(bytes) });
                   }
                   // else: orphan image, drop
               }
           }
           if let Some(prev) = pending_xml.take() {
               out.push(EventPair { xml: prev, jpeg: None });
           }
           Ok(out)
       }

       /// Fallback parser for bodies without Content-Disposition headers.
       /// Scans for <EventNotificationAlert>...</EventNotificationAlert> delimiters.
       pub fn parse_line_scan_fallback(body: &[u8]) -> Vec<EventPair> {
           // Find each <EventNotificationAlert>...</EventNotificationAlert> block as XML.
           // JPEG attachment: scan immediately following bytes for \xFF\xD8\xFF SOI marker up to the next boundary or `<EventNotificationAlert`.
           let mut out = Vec::new();
           let open = b"<EventNotificationAlert";
           let close = b"</EventNotificationAlert>";
           let mut start = 0usize;
           while let Some(o) = find_subslice(&body[start..], open) {
               let abs_o = start + o;
               if let Some(c) = find_subslice(&body[abs_o..], close) {
                   let abs_c = abs_o + c + close.len();
                   let xml_bytes = Bytes::copy_from_slice(&body[abs_o..abs_c]);
                   // Look for JPEG SOI within next 256 bytes of slack + up to next open or EOF
                   let tail_start = abs_c;
                   let tail_end = find_subslice(&body[tail_start..], open).map(|n| tail_start + n).unwrap_or(body.len());
                   let jpeg = find_subslice(&body[tail_start..tail_end], b"\xFF\xD8\xFF").map(|j| {
                       Bytes::copy_from_slice(&body[tail_start + j..tail_end])
                   });
                   out.push(EventPair { xml: xml_bytes, jpeg });
                   start = tail_end;
               } else {
                   break;
               }
           }
           out
       }

       fn find_subslice(h: &[u8], n: &[u8]) -> Option<usize> {
           h.windows(n.len()).position(|w| w == n)
       }
       ```

       ```rust
       }

       pub async fn connect_and_stream(cfg: &DeviceConfig, state: &AppState) -> anyhow::Result<()> {
           let url = format!("{}/ISAPI/Event/notification/alertStream", cfg.base_url);
           let client = Client::builder()
               .connect_timeout(Duration::from_secs(5))
               .danger_accept_invalid_certs(cfg.allow_insecure_tls)
               .build()?;

           let resp = client
               .get(&url)
               .send_digest_auth((cfg.username.as_str(), cfg.password.as_str()))
               .await?;
           anyhow::ensure!(resp.status().is_success(), "bad status {}", resp.status());

           let content_type = resp
               .headers()
               .get(reqwest::header::CONTENT_TYPE)
               .and_then(|v| v.to_str().ok())
               .ok_or_else(|| anyhow::anyhow!("missing content-type"))?
               .to_string();

           let stream = resp.bytes_stream();
           // T-2-19 mitigation (ASVS L1 V12): bound per-field + whole-stream sizes.
           // whole_stream is set generously (1 GiB) because alertStream is long-lived;
           // per_field is the real DoS cap and matches parser::parse_buffer.
           let constraints = multer::Constraints::new()
               .size_limit(
                   multer::SizeLimit::new()
                       .per_field(10 * 1024 * 1024)            // 10 MB per part (JPEG ceiling)
                       .whole_stream(1024 * 1024 * 1024),      // 1 GiB over the entire session
               );
           let mut mp = multer::Multipart::with_constraints(stream, boundary, constraints);

           // On first byte successfully read, mark device online + touch last_seen.
           super::super::supervisor::status::update_connection_state(state, &cfg.id, "online").await?;
           touch_last_seen(state, &cfg.id).await?;

           let mut pending_xml: Option<Bytes> = None;
           let mut pending_xml_raw: Option<String> = None;

           while let Some(field_res) = mp.next_field().await.transpose() {
               let field = field_res?;
               let ct = field.content_type().map(|m| m.to_string()).unwrap_or_default();
               let bytes = field.bytes().await?;

               if ct.starts_with("application/xml") || bytes.starts_with(b"<EventNotificationAlert") {
                   if let Some(prev_xml) = pending_xml.take() {
                       let raw = pending_xml_raw.take().unwrap_or_default();
                       ingest_pair(state, cfg, EventPair { xml: prev_xml, jpeg: None }, raw).await?;
                   }
                   touch_last_seen(state, &cfg.id).await?;
                   pending_xml_raw = Some(std::str::from_utf8(&bytes).unwrap_or_default().to_string());
                   pending_xml = Some(bytes);

                       let raw = pending_xml_raw.take().unwrap_or_default();
                       ingest_pair(state, cfg, EventPair { xml: prev_xml, jpeg: Some(bytes) }, raw).await?;
                   }
                   // orphan image: drop
               }
           }

               let raw = pending_xml_raw.take().unwrap_or_default();
               ingest_pair(state, cfg, EventPair { xml: prev_xml, jpeg: None }, raw).await?;
           }
           Ok(())
       }

       async fn ingest_pair(state: &AppState, cfg: &DeviceConfig, pair: EventPair, raw_xml: String) -> anyhow::Result<()> {
           let xml_str = std::str::from_utf8(&pair.xml)?;
           let stripped = strip_xmlns(xml_str);
           let alert: EventNotificationAlert = quick_xml::de::from_str(&stripped)?;
           if alert.is_heartbeat() {
               tracing::debug!(device_id=%cfg.id, "heartbeat received");
               return Ok(());
           }
           let Some(ace) = alert.access_controller_event.as_ref() else {
               tracing::debug!(device_id=%cfg.id, "alert without AccessControllerEvent — skipped");
               return Ok(());
           };
           let captured_at = alert.captured_at_epoch().unwrap_or_else(|| chrono::Utc::now().timestamp());
           let direction = if !ace.attendance_status.is_empty() {
               direction_for_attendance_status(&ace.attendance_status).to_string()
           } else {
               cfg.direction_default.clone()
           };

           let conn = state.db.connect().map_err(anyhow::Error::from)?;
           let face_id = if ace.face_id.is_empty() { None } else { Some(ace.face_id.clone()) };
           let employee_no_string = if ace.employee_no_string.is_empty() { None } else { Some(ace.employee_no_string.clone()) };
           let employee_id = events_service::lookup_employee_for_event(&conn, &cfg.id, face_id.as_deref(), employee_no_string.as_deref())
               .await
               .map_err(|e| anyhow::anyhow!("lookup error: {e}"))?;
           let is_unknown = employee_id.is_none();

           let new_event = NewAttendanceEvent {
               id: uuid::Uuid::new_v4().to_string(),
               employee_id,
               device_id: cfg.id.clone(),
               direction,
               captured_at,
               is_unknown,
               face_id,
               employee_no_string,
               raw_xml,
               photo_bytes: pair.jpeg.map(|b| b.to_vec()),
           };

           match events_service::persist_attendance_event(&conn, new_event).await {
               Ok(PersistOutcome::Inserted { .. }) => {
                   tracing::info!(device_id=%cfg.id, "event persisted");
               }
               Ok(PersistOutcome::Deduplicated) => {
                   tracing::debug!(device_id=%cfg.id, "event deduplicated");
               }
               Err(e) => {
                   tracing::warn!(device_id=%cfg.id, error=?e, "persist failed");
                   return Err(anyhow::anyhow!("persist failed: {e}"));
               }
           }
           Ok(())
       }
       ```

       ```rust
       use std::collections::HashMap;
       use std::sync::Arc;
       use tokio::sync::{mpsc, Mutex};
       use tokio::task::JoinHandle;
       use tokio_util::sync::CancellationToken;

       use crate::state::AppState;
       use crate::devices::service as devices_service;

       pub mod status;
       pub mod task;
       pub mod watchdog;

       #[derive(Debug, Clone)]
       pub enum DeviceLifecycleEvent {
           Start(String),
           Stop(String),
           Restart(String),
       }

       pub type LifecycleTx = mpsc::UnboundedSender<DeviceLifecycleEvent>;

       pub struct Supervisor {
           state: AppState,
           shutdown: CancellationToken,
           handles: Arc<Mutex<HashMap<String, (JoinHandle<()>, CancellationToken)>>>,
       }

       impl Supervisor {
           pub fn new(state: AppState, shutdown: CancellationToken) -> Self {
               Self { state, shutdown, handles: Arc::new(Mutex::new(HashMap::new())) }
           }

           /// Main loop: bootstrap from DB, then process lifecycle events until shutdown.
           pub async fn run(self, mut lifecycle_rx: mpsc::UnboundedReceiver<DeviceLifecycleEvent>) {
               // Bootstrap
               if let Ok(conn) = self.state.db.connect() {
                   if let Ok(devices) = devices_service::list_active(&conn, &self.state.config.device_creds_key).await {
                       for dev in devices { self.spawn_device(dev).await; }
                   }
               }

               // Process events until shutdown
               loop {
                   tokio::select! {
                       biased;
                       _ = self.shutdown.cancelled() => break,
                       Some(ev) = lifecycle_rx.recv() => self.handle_event(ev).await,
                   }
               }

               // Graceful shutdown: cancel all children, await joins
               let handles: Vec<_> = {
                   let mut h = self.handles.lock().await;
                   let drained = std::mem::take(&mut *h);
                   drained.into_iter().collect()
               };
               for (_, (_, tok)) in &handles { tok.cancel(); }
               for (_, (handle, _)) in handles { let _ = handle.await; }
           }

           async fn spawn_device(&self, dev: devices_service::DeviceWithPlaintext) {
               let tok = self.shutdown.child_token();
               let state = self.state.clone();
               let dev_id = dev.id.clone();
               let tok_clone = tok.clone();
               let handle = tokio::spawn(async move {
                   task::device_task(dev, state, tok_clone).await;
               });
               self.handles.lock().await.insert(dev_id, (handle, tok));
           }

           async fn handle_event(&self, ev: DeviceLifecycleEvent) {
               match ev {
                   DeviceLifecycleEvent::Start(id) => { /* load + spawn if not present */ }
                   DeviceLifecycleEvent::Stop(id) => { /* cancel token + await join + remove */ }
                   DeviceLifecycleEvent::Restart(id) => { /* Stop then Start */ }
               }
           }

           #[cfg(test)]
           pub async fn active_count(&self) -> usize {
               self.handles.lock().await.len()
           }
       }
       ```

       ```rust
       use crate::state::AppState;

       pub async fn device_task(dev: DeviceWithPlaintext, state: AppState, cancel: CancellationToken) {
           let cfg = DeviceConfig {
               id: dev.id.clone(),
               base_url: dev.base_url.clone(),
               username: dev.username.clone(),
               password: dev.password.clone(),  // local to this task frame — NEVER logged
               direction_default: dev.direction.clone(),
               allow_insecure_tls: dev.allow_insecure_tls,
           };
           let mut backoff_ms: u64 = 1_000;
           loop {
               tokio::select! {
                   biased;
                   _ = cancel.cancelled() => { crate::supervisor::status::update_connection_state(&state, &cfg.id, "offline").await.ok(); return; }
                   res = connect_and_stream(&cfg, &state) => {

                           crate::supervisor::status::update_connection_state(&state, &cfg.id, "offline").await.ok();
                       } else {
                           // Graceful end — reset backoff
                           tracing::info!(device_id=%cfg.id, "stream ended gracefully");
                           backoff_ms = 1_000;
                       }
                       let jitter_ms: u64 = {
                           let j_max = backoff_ms / 4;
                           if j_max == 0 { 0 } else { rand::thread_rng().gen_range(0..=j_max) }
                       };
                       let sleep_ms = backoff_ms.saturating_add(jitter_ms);
                       tokio::select! {
                           _ = cancel.cancelled() => { return; }
                           _ = tokio::time::sleep(Duration::from_millis(sleep_ms)) => {}
                       }
                       backoff_ms = (backoff_ms.saturating_mul(2)).min(60_000);
                   }
               }
           }
       ```
