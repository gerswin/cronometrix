# Live CI Gates and Branch Protection Validation Plan

> **For agentic workers:** REQUIRED SUB-SKILLS: use `superpowers:executing-plans`, `superpowers:test-driven-development` for the protection normalizer, and `superpowers:verification-before-completion` before every evidence commit. All GitHub mutations are deliberate validation actions and must be linked in the summary.

**Goal:** prove that the exact Phase 12 candidate passes every required check, apply and verify non-weakening protection on `main`, and prove deliberate regressions are rejected by both CI and branch protection.

**Architecture:** validate an immutable candidate PR first; capture, normalize, apply, and re-read branch protection second; only then create the deliberate red PR. Close and delete the red branch without merge. Evidence contains URLs plus redacted JSON/text, never credentials or source-bearing reports.

**Tech Stack:** GitHub Actions, GitHub CLI/API, git, jq, Python unittest.

## Global Constraints

- Begin only after `12-05-SUMMARY.md` says `Verdict: PASS`.
- Set `CANDIDATE_SHA` once from remote branch `codex/v1-release-candidate`; never create, move, or force-update that branch here.
- Commit validation evidence only on `codex/v1-live-validation-evidence-*`.
- If a source commit changes, stop and restart candidate validation.
- If `origin/main` moves after `BASE_SHA` is captured, stop; remediate and repeat Phase 12/13 on a new candidate rather than validating an untested composition.
- Never merge the deliberate red PR. Close it and delete its remote branch.
- Never disable or soften a CI threshold or protection setting.
- Preserve private Actions reports in GitHub for 14 days; commit only safe fields such as URLs, conclusions, names, SHAs, timestamps, and hashes.
- Branch protection is an external mutation. Capture the prior configuration and preserve every stricter existing setting.
- If existing `required_linear_history` or `lock_branch` is enabled, stop with `Verdict: BLOCKED`; the approved merge/promotion flow must be reconciled explicitly instead of silently disabling either setting.
- Treat every code fence as a fresh Bash process. Each block starts `set -euo pipefail` and sources `/tmp/cronometrix-v1-validation.env`; Task 1 creates that mode-0600 file with non-secret immutable identities and later steps append only non-secret paths/URLs.

Only after the red proof, create
`scripts/normalize-branch-protection.py` with this exact side-effect-free
implementation. It follows GitHub's current GET/PUT schema for status-check
`checks`, actor restrictions, review settings, and wrapped booleans:

```python
#!/usr/bin/env python3
from copy import deepcopy
import json
from pathlib import Path
import sys

REQUIRED_CHECKS = (
    "Backend Coverage",
    "Frontend Coverage",
    "E2E Tests",
    "Container Smoke",
    "Release Gate",
)

_WRAPPED_BOOLEANS = (
    "enforce_admins",
    "required_linear_history",
    "allow_force_pushes",
    "allow_deletions",
    "block_creations",
    "required_conversation_resolution",
    "lock_branch",
    "allow_fork_syncing",
)
_ACTOR_FIELDS = {"users": "login", "teams": "slug", "apps": "slug"}


def _enabled(value: object) -> bool:
    if isinstance(value, dict):
        return bool(value.get("enabled", False))
    return bool(value)


def _actor_names(items: object, field: str) -> list[str]:
    if items is None:
        return []
    if not isinstance(items, list):
        raise ValueError(f"actor list for {field} must be a list")
    result: list[str] = []
    for item in items:
        value = item if isinstance(item, str) else item.get(field) if isinstance(item, dict) else None
        if not isinstance(value, str) or not value:
            raise ValueError(f"actor entry for {field} has no stable name")
        if value not in result:
            result.append(value)
    return result


def _actors(block: object) -> dict[str, list[str]] | None:
    if block is None:
        return None
    if not isinstance(block, dict):
        raise ValueError("actor restriction must be an object or null")
    return {
        key: _actor_names(block.get(key, []), field)
        for key, field in _ACTOR_FIELDS.items()
    }


def _status(block: object) -> dict | None:
    if block is None:
        return None
    if not isinstance(block, dict):
        raise ValueError("required_status_checks must be an object or null")
    raw_contexts = block.get("contexts", [])
    if not isinstance(raw_contexts, list) or not all(isinstance(item, str) for item in raw_contexts):
        raise ValueError("status contexts must be strings")
    contexts = list(dict.fromkeys(raw_contexts))
    checks: list[dict] = []
    for raw in block.get("checks", []) or []:
        if not isinstance(raw, dict) or not isinstance(raw.get("context"), str):
            raise ValueError("each status check needs a context")
        check = {"context": raw["context"]}
        app_id = raw.get("app_id")
        if app_id is not None:
            if isinstance(app_id, bool) or not isinstance(app_id, int):
                raise ValueError("status check app_id must be an integer or null")
            check["app_id"] = app_id
        if check not in checks:
            checks.append(check)
    return {"strict": bool(block.get("strict", False)), "contexts": contexts, "checks": checks}


def _reviews(block: object) -> dict | None:
    if block is None:
        return None
    if not isinstance(block, dict):
        raise ValueError("required_pull_request_reviews must be an object or null")
    count = block.get("required_approving_review_count", 0)
    if isinstance(count, bool) or not isinstance(count, int) or not 0 <= count <= 6:
        raise ValueError("required_approving_review_count must be between 0 and 6")
    result = {
        "dismiss_stale_reviews": bool(block.get("dismiss_stale_reviews", False)),
        "require_code_owner_reviews": bool(block.get("require_code_owner_reviews", False)),
        "required_approving_review_count": count,
        "require_last_push_approval": bool(block.get("require_last_push_approval", False)),
    }
    for key in ("dismissal_restrictions", "bypass_pull_request_allowances"):
        if key in block and block[key] is not None:
            result[key] = _actors(block[key])
    return result


def _as_put_payload(source: dict) -> dict:
    if not isinstance(source, dict):
        raise ValueError("branch protection input must be an object")
    source = deepcopy(source)
    payload = {
        "required_status_checks": _status(source.get("required_status_checks")),
        "enforce_admins": _enabled(source.get("enforce_admins")),
        "required_pull_request_reviews": _reviews(source.get("required_pull_request_reviews")),
        "restrictions": _actors(source.get("restrictions")),
    }
    for key in _WRAPPED_BOOLEANS[1:]:
        payload[key] = _enabled(source.get(key))
    return payload


def normalize(source: dict) -> dict:
    """Return a non-weakening GitHub PUT payload without mutating source."""
    payload = _as_put_payload(source)
    for blocker in ("required_linear_history", "lock_branch"):
        if payload[blocker]:
            raise ValueError(f"{blocker} is enabled; stop and reconcile the release flow")

    status = payload["required_status_checks"] or {"strict": False, "contexts": [], "checks": []}
    status["strict"] = True
    present = set(status["contexts"])
    present.update(check["context"] for check in status["checks"])
    for name in REQUIRED_CHECKS:
        if name not in present:
            status["contexts"].append(name)
            present.add(name)
    payload["required_status_checks"] = status

    reviews = payload["required_pull_request_reviews"] or {
        "dismiss_stale_reviews": False,
        "require_code_owner_reviews": False,
        "required_approving_review_count": 0,
        "require_last_push_approval": False,
    }
    reviews["dismiss_stale_reviews"] = True
    reviews["required_approving_review_count"] = max(1, reviews["required_approving_review_count"])
    payload["required_pull_request_reviews"] = reviews

    payload["enforce_admins"] = True
    payload["allow_force_pushes"] = False
    payload["allow_deletions"] = False
    payload["required_conversation_resolution"] = True
    return payload


def _require(condition: bool, message: str) -> None:
    if not condition:
        raise ValueError(message)


def _require_actor_subset(expected: dict | None, actual: dict | None, label: str) -> None:
    if expected is None or not any(expected.values()):
        return
    _require(actual is not None, f"{label} disappeared")
    assert actual is not None
    for key in _ACTOR_FIELDS:
        _require(set(expected[key]) <= set(actual[key]), f"{label}.{key} was weakened")


def verify_effective(before: dict, desired: dict, after: dict) -> None:
    """Raise if the live GET response is weaker than normalize(before)."""
    expected = normalize(before)
    _require(desired == expected, "desired payload is not normalize(before)")
    actual = _as_put_payload(after)

    expected_status = expected["required_status_checks"]
    actual_status = actual["required_status_checks"]
    _require(actual_status is not None and actual_status["strict"], "strict status checks are not effective")
    assert actual_status is not None
    expected_names = set(expected_status["contexts"])
    expected_names.update(check["context"] for check in expected_status["checks"])
    actual_names = set(actual_status["contexts"])
    actual_names.update(check["context"] for check in actual_status["checks"])
    _require(expected_names <= actual_names, "required status checks were dropped")
    actual_bound = {
        (check["context"], check.get("app_id"))
        for check in actual_status["checks"]
        if "app_id" in check
    }
    for check in expected_status["checks"]:
        if "app_id" in check:
            _require(
                (check["context"], check["app_id"]) in actual_bound,
                f"GitHub App binding changed for {check['context']}",
            )

    expected_reviews = expected["required_pull_request_reviews"]
    actual_reviews = actual["required_pull_request_reviews"]
    _require(actual_reviews is not None, "pull-request reviews are not required")
    assert actual_reviews is not None
    _require(
        actual_reviews["required_approving_review_count"]
        >= expected_reviews["required_approving_review_count"],
        "required review count was weakened",
    )
    for key in ("dismiss_stale_reviews", "require_code_owner_reviews", "require_last_push_approval"):
        if expected_reviews[key]:
            _require(actual_reviews[key], f"{key} was weakened")
    for key in ("dismissal_restrictions", "bypass_pull_request_allowances"):
        _require_actor_subset(expected_reviews.get(key), actual_reviews.get(key), key)

    _require_actor_subset(expected["restrictions"], actual["restrictions"], "restrictions")
    for key in ("enforce_admins", "block_creations", "required_conversation_resolution"):
        if expected[key]:
            _require(actual[key], f"{key} was weakened")
    for key in ("allow_force_pushes", "allow_deletions", "required_linear_history", "lock_branch"):
        _require(actual[key] is False, f"{key} must remain false")
    _require(
        actual["allow_fork_syncing"] == expected["allow_fork_syncing"],
        "allow_fork_syncing changed",
    )
    if _enabled(before.get("required_signatures")):
        _require(_enabled(after.get("required_signatures")), "required_signatures was weakened")


def _read(path: str) -> dict:
    return json.loads(Path(path).read_text())


def main(argv: list[str] | None = None) -> int:
    args = list(sys.argv[1:] if argv is None else argv)
    try:
        if len(args) == 1:
            print(json.dumps(normalize(_read(args[0])), indent=2, sort_keys=True))
        elif len(args) == 4 and args[0] == "--verify":
            verify_effective(_read(args[1]), _read(args[2]), _read(args[3]))
            print(json.dumps({"verified": True}))
        else:
            raise ValueError("usage: normalize-branch-protection.py [--verify BEFORE DESIRED AFTER] INPUT")
    except (OSError, json.JSONDecodeError, ValueError) as error:
        print(str(error), file=sys.stderr)
        return 2
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
