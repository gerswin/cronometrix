    import { test, expect } from '@playwright/test'
    import { API_BASE } from './fixtures/api'

    // RBAC source of truth: backend/src/main.rs route_layer assignments.
    // As of Phase 9 planning the contract is approximately:
    //   - viewer_routes:        GET /employees, GET /departments, GET /devices, GET /timesheet
    //                           (read-only)
    //   - supervisor_read_routes: GET /audit, plus all viewer reads
    //   - supervisor_routes:    POST /employees (create), PATCH /timesheet (overrides),
    //                           POST /leaves, POST /devices/{id}/commands (door open) —
    //                           Supervisor + Admin only
    //   - admin_routes:         DELETE /employees, DELETE /devices, system config
    //                           Admin only
    //
    // BEFORE COMMITTING: read backend/src/main.rs and verify each test's expected
    // status matches the actual route_layer assignment. If main.rs has changed,
    // update the test expectations and add a comment noting the change.

    test.describe('RBAC cross-cut (D-01 + cross-stack)', () => {
      // Helper: build a request context with a specific role's storageState
      async function withRole(browser: import('@playwright/test').Browser, role: 'admin'|'supervisor'|'viewer') {
        const ctx = await browser.newContext({ storageState: `e2e/.auth/${role}.json` })
        return { ctx, request: ctx.request }
      }

      test('viewer: GET /employees → 200 (read allowed per viewer_routes)', async ({ browser }) => {
        const { ctx, request } = await withRole(browser, 'viewer')
        const r = await request.get(`${API_BASE}/employees`)
        expect(r.status()).toBe(200)
        await ctx.close()
      })

      test('viewer: POST /employees → 403 (supervisor+ only)', async ({ browser }) => {
        // Verify against main.rs: POST /employees is in supervisor_routes (NOT viewer_routes)
        const { ctx, request } = await withRole(browser, 'viewer')
        const r = await request.post(`${API_BASE}/employees`, {
          data: { name: 'should not work', employee_code: 'X1', department_id: 'dept-prod' },
        })
        expect(r.status()).toBe(403)
        await ctx.close()
      })

      test('viewer: POST /devices/{id}/commands → 403', async ({ browser }) => {
        // Verify against main.rs: device commands route is supervisor+
        const { ctx, request } = await withRole(browser, 'viewer')
        const r = await request.post(`${API_BASE}/devices/dev-entry/commands`, {
          data: { command: 'door_open' },
        })
        expect(r.status()).toBe(403)
        await ctx.close()
      })

      test('viewer: POST /leaves → 403 (supervisor+ only)', async ({ browser }) => {
        // Verify against main.rs: leaves create is supervisor_routes
        const { ctx, request } = await withRole(browser, 'viewer')
        const r = await request.post(`${API_BASE}/leaves`, {
          data: { employee_id: 'emp-ana', leave_type: 'medical', from_date: '2026-04-15', to_date: '2026-04-16', justification: 'x' },
        })
        expect(r.status()).toBe(403)
        await ctx.close()
      })

      test('supervisor: POST /employees → not 403 (supervisor_routes allows create)', async ({ browser }) => {
        // Verify against main.rs: supervisor_routes includes POST /employees
        const { ctx, request } = await withRole(browser, 'supervisor')
        const r = await request.post(`${API_BASE}/employees`, {
          data: { name: 'Supervisor Created', employee_code: 'SUP001', department_id: 'dept-prod' },
        })
        // 200/201 acceptable; 422 for validation also acceptable as long as NOT 403
        expect(r.status()).not.toBe(403)
        await ctx.close()
      })

      test('supervisor: DELETE /employees/{id} → 403 (admin-only per admin_routes)', async ({ browser }) => {
        // Verify against main.rs: DELETE /employees is in admin_routes (not supervisor_routes)
        // If main.rs allows supervisor delete, update this test to 200/204 and document why.
        const { ctx, request } = await withRole(browser, 'supervisor')
        const r = await request.delete(`${API_BASE}/employees/emp-ana`)
        expect(r.status()).toBe(403)
        await ctx.close()
      })

      test('admin: full access — POST /employees + DELETE /employees/{id}', async ({ browser }) => {
        const { ctx, request } = await withRole(browser, 'admin')
        // Create
        const create = await request.post(`${API_BASE}/employees`, {
          data: { name: 'Admin Created Then Deleted', employee_code: 'ADC001', department_id: 'dept-prod' },
        })
        expect([200, 201]).toContain(create.status())
        // Delete
        const body = await create.json().catch(() => ({}))
        if (body?.id) {
          const del = await request.delete(`${API_BASE}/employees/${body.id}`)
          expect([200, 204]).toContain(del.status())
        }
        await ctx.close()
      })

      test('unauthenticated: GET /employees → 401', async ({ browser }) => {
        const ctx = await browser.newContext()   // no storageState
        const request = ctx.request
        const r = await request.get(`${API_BASE}/employees`)
        expect(r.status()).toBe(401)
        await ctx.close()
      })

        const page = await ctx.newPage()
        await page.goto('/employees')
        await expect(page.getByTestId('new-employee-button')).toHaveCount(0)
        await ctx.close()
      })
    })
