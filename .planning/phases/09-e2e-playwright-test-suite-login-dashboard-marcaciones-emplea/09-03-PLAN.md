    //!
    //! Reuses production code paths so password hashes match the login flow:
    //! - cronometrix_api::config::Config::from_env (reads TURSO_DATABASE_URL)
    //! - cronometrix_api::db::init_db (runs all migrations)
    //! - cronometrix_api::auth::service::hash_password (argon2id, identical params)
    //!
    //! Refuses to run without CRONOMETRIX_E2E=true. Idempotent — INSERT OR IGNORE.

    use cronometrix_api::{auth, config::Config, db};

    #[tokio::main]
    async fn main() -> anyhow::Result<()> {
        if std::env::var("CRONOMETRIX_E2E").as_deref() != Ok("true") {
            eprintln!("seed_e2e refuses to run without CRONOMETRIX_E2E=true");
            std::process::exit(2);
        }

        dotenvy::dotenv().ok();
        tracing_subscriber::fmt()
            .with_env_filter(std::env::var("RUST_LOG").unwrap_or_else(|_| "info".into()))
            .init();

        let config = Config::from_env()?;
        let db = db::init_db(&config).await?;
        let conn = db.connect()?;

        // ----- Users -----
        let admin_hash = auth::service::hash_password("e2e-admin-pass")?;
        let supervisor_hash = auth::service::hash_password("e2e-supervisor-pass")?;
        let viewer_hash = auth::service::hash_password("e2e-viewer-pass")?;

        for (id, username, role, hash) in [
            ("e2e-admin-id", "e2e_admin", "admin", &admin_hash),
            ("e2e-supervisor-id", "e2e_supervisor", "supervisor", &supervisor_hash),
            ("e2e-viewer-id", "e2e_viewer", "viewer", &viewer_hash),
        ] {
            // NOTE: column names below MUST match backend/src/db/migrations/001_initial_schema.sql
            // — read that migration before assuming. If users.password_hash differs, fix here.
            conn.execute(
                "INSERT OR IGNORE INTO users (id, username, role, password_hash, created_at)
                 VALUES (?1, ?2, ?3, ?4, strftime('%s','now'))",
                libsql::params![id, username, role, hash.as_str()],
            ).await?;
        }

        // ----- Departments (3 — generous coverage for filter tests) -----
        for (id, name) in [
            ("dept-prod", "Producción"),
            ("dept-admin", "Administración"),
            ("dept-rrhh", "Recursos Humanos"),

                "INSERT OR IGNORE INTO departments
                 (id, name, base_salary, work_start, work_end, lunch_mode, lunch_minutes, created_at)
                 VALUES (?1, ?2, 1000.0, '08:00', '17:00', 'fixed', 60, strftime('%s','now'))",
                libsql::params![id, name],
            ).await?;
        }

        // ----- Employees (6 — 2 per department, varied names for filter tests) -----
        let employees = [
            ("emp-ana", "EMP001", "Ana Pérez", "dept-prod"),
            ("emp-luis", "EMP002", "Luis García", "dept-prod"),
            ("emp-maria", "EMP003", "María López", "dept-admin"),
            ("emp-pedro", "EMP004", "Pedro Ramírez", "dept-admin"),
            ("emp-carmen", "EMP005", "Carmen Silva", "dept-rrhh"),
            ("emp-jose", "EMP006", "José Hernández", "dept-rrhh"),
        ];
        for (id, code, name, dept_id) in employees {
            conn.execute(
                "INSERT OR IGNORE INTO employees
                 (id, employee_code, name, department_id, status, created_at)
                 VALUES (?1, ?2, ?3, ?4, 'active', strftime('%s','now'))",
                libsql::params![id, code, name, dept_id],
            ).await?;

    /// Handle a PUT command, record it in recv_log (B6), return canned 200 XML.
    async fn handle_xml_ok_recorded(
        State(state): State<MockState>,
        req: axum::http::Request<axum::body::Body>,
    ) -> Response {
        let method = req.method().to_string();
        let path = req.uri().path().to_string();
        let body_bytes = axum::body::to_bytes(req.into_body(), 64 * 1024).await
            .unwrap_or_default();
        let body = String::from_utf8_lossy(&body_bytes).to_string();
        let ts = std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH).unwrap().as_millis();
        state.recv_log.lock().await.push(ReceivedCall {
            method, path, body, timestamp_ms: ts,
        });
        let resp_body = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n\
                         <ResponseStatus><statusCode>1</statusCode><statusString>OK</statusString></ResponseStatus>";
        ([(axum::http::header::CONTENT_TYPE, "application/xml")], resp_body).into_response()
    }

    /// Drain queue, format as multipart/mixed (matches backend/src/isapi/stream.rs parser).
    async fn handle_alert_stream(State(state): State<MockState>) -> Response {
        let mut queue = state.event_queue.lock().await;
        let chunks: Vec<String> = std::mem::take(&mut *queue);
        drop(queue);
        let boundary = "MIME_boundary";
        let mut body = String::new();
        for xml in chunks {
            body.push_str(&format!("--{}\r\n", boundary));
            body.push_str("Content-Type: application/xml\r\n\r\n");
            body.push_str(&xml);
            body.push_str("\r\n");
        }
        body.push_str(&format!("--{}--\r\n", boundary));
        ([(axum::http::header::CONTENT_TYPE,
           format!("multipart/mixed; boundary={}", boundary).as_str())], body)
            .into_response()
    }

    #[derive(Deserialize)]
    struct PushEvent { xml: String }

    async fn handle_push_event(
        State(state): State<MockState>,
        Json(payload): Json<PushEvent>,
    ) -> Json<serde_json::Value> {
        state.event_queue.lock().await.push(payload.xml);
        Json(serde_json::json!({ "queued": true }))
    }

    async fn handle_clear_queue(State(state): State<MockState>) -> Json<serde_json::Value> {
        state.event_queue.lock().await.clear();
        Json(serde_json::json!({ "cleared": true }))
    }

    /// B6 — return the list of every PUT/POST received on the public surface.
    async fn handle_recv_log(State(state): State<MockState>) -> Json<serde_json::Value> {
        let log = state.recv_log.lock().await.clone();
        Json(serde_json::json!({ "commands": log }))
    }

    async fn handle_clear_recv_log(State(state): State<MockState>) -> Json<serde_json::Value> {
        state.recv_log.lock().await.clear();
        Json(serde_json::json!({ "cleared": true }))
    }
