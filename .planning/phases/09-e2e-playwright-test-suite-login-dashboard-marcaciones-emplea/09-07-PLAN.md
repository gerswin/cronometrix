    ```ts
    import { test, expect } from '@playwright/test'
    import { API_BASE } from './fixtures/api'

    // No test.use({ storageState }) — login spec uses fresh contexts.

    test.describe('Login (UAT depth, English copy per Addendum D-19)', () => {
      test('renders login form with English labels', async ({ page }) => {
        await page.goto('/login')
        await expect(page.getByLabel('Username')).toBeVisible()
        await expect(page.getByLabel('Password')).toBeVisible()
        await expect(page.getByRole('button', { name: 'Log in' })).toBeVisible()
      })

      test('happy path: admin logs in and lands on /', async ({ page }) => {
        await page.goto('/login')
        await page.getByLabel('Username').fill('e2e_admin')
        await page.getByLabel('Password').fill('e2e-admin-pass')
        await page.getByRole('button', { name: 'Log in' }).click()
        await expect(page).toHaveURL(/\/$|\/dashboard/)   // landing route
      })

      test('invalid credentials → 401 → "Invalid username or password."', async ({ page }) => {
        await page.goto('/login')
        await page.getByLabel('Username').fill('e2e_admin')
        await page.getByLabel('Password').fill('wrong-password')
        await page.getByRole('button', { name: 'Log in' }).click()
        await expect(page.getByText('Invalid username or password.')).toBeVisible()
        await expect(page).toHaveURL(/\/login/)
      })

      test('validation: empty username shows form error', async ({ page }) => {
        await page.goto('/login')
        await page.getByRole('button', { name: 'Log in' }).click()
        // Either the FormMessage shows zod error OR the button never submits.
        // Assert the user is still on /login.
        await expect(page).toHaveURL(/\/login/)
        // Check for ANY validation message rendered by react-hook-form/zod.
        const errors = page.locator('[role="alert"], .text-destructive, .text-red-500, .text-rose-500')
        await expect(errors.first()).toBeVisible({ timeout: 3000 }).catch(() => {})
      })

      test('validation: short password rejected by zod schema', async ({ page }) => {
        await page.goto('/login')
        await page.getByLabel('Username').fill('e2e_admin')
        await page.getByLabel('Password').fill('a')   // assume schema requires >= 8
        await page.getByRole('button', { name: 'Log in' }).click()
        await expect(page).toHaveURL(/\/login/)
      })

      test('toggle password visibility (Eye/EyeOff icon)', async ({ page }) => {
        await page.goto('/login')
        const pwd = page.getByLabel('Password')
        await pwd.fill('secret123')
        await expect(pwd).toHaveAttribute('type', 'password')
        // Click the eye toggle (no accessible name guaranteed — use the icon button location)
        const toggle = page.locator('button[type="button"]').filter({ has: page.locator('svg') }).first()
        await toggle.click()
        await expect(pwd).toHaveAttribute('type', 'text')
      })

      test('session persists across refresh', async ({ page }) => {
        await page.goto('/login')
        await page.getByLabel('Username').fill('e2e_admin')
        await page.getByLabel('Password').fill('e2e-admin-pass')
        await page.getByRole('button', { name: 'Log in' }).click()
        await expect(page).toHaveURL(/\/$|\/dashboard/)
        await page.reload()
        // Should still be authenticated (refresh-cookie + refresh hook re-issues access_token)
        await expect(page).not.toHaveURL(/\/login/)
      })

      test('multi-tab: second tab shares session', async ({ context, page }) => {
        await page.goto('/login')
        await page.getByLabel('Username').fill('e2e_admin')
        await page.getByLabel('Password').fill('e2e-admin-pass')
        await page.getByRole('button', { name: 'Log in' }).click()
        await expect(page).toHaveURL(/\/$|\/dashboard/)
        const tab2 = await context.newPage()
        await tab2.goto('/employees')
        await expect(tab2).not.toHaveURL(/\/login/)
      })

      test('RBAC redirect: viewer cannot reach /devices (UI hidden OR 403 toast)', async ({ page }) => {
        await page.goto('/login')
        await page.getByLabel('Username').fill('e2e_viewer')
        await page.getByLabel('Password').fill('e2e-viewer-pass')
        await page.getByRole('button', { name: 'Log in' }).click()
        await expect(page).toHaveURL(/\/$|\/dashboard/)
        await page.goto('/devices')
        // Per Phase 4 D-14: ISAPI command buttons hidden for viewer.
        // Assert: either redirect away from /devices OR no admin commands visible.
        // The viewer GET /devices is permitted (read-only); but POST /devices/.../commands → 403.
        const adminButtons = page.getByRole('button', { name: /Abrir puerta|Reiniciar|Modo enroll/i })
        await expect(adminButtons).toHaveCount(0)
      })

      test('redirect param honored after login', async ({ page }) => {
        await page.goto('/login?redirect=/employees')
        await page.getByLabel('Username').fill('e2e_admin')
        await page.getByLabel('Password').fill('e2e-admin-pass')
        await page.getByRole('button', { name: 'Log in' }).click()
        await expect(page).toHaveURL(/\/employees/)
      })

      test('redirect param sanitized: "//evil.com" falls back to "/"', async ({ page }) => {
        await page.goto('/login?redirect=//evil.com')
        await page.getByLabel('Username').fill('e2e_admin')
        await page.getByLabel('Password').fill('e2e-admin-pass')
        await page.getByRole('button', { name: 'Log in' }).click()
        // safeRedirect in login/page.tsx must reject this and route to '/'.
        await expect(page).not.toHaveURL(/evil\.com/)
      })
    })
    ```
