       ```ts
       export const CARACAS_TZ = 'America/Caracas'

       /**
        * Convert ISO date + HH:mm (Caracas local) to UTC epoch seconds.
        * Caracas = UTC-4, no DST since 2016, so the conversion is constant.
        */
       export function caracasEpoch(isoDate: string, hhmm: string): number {
         // Caracas = UTC-4 fixed; isoDate "2026-04-15" + "08:00" = "2026-04-15T12:00:00Z"
         const [h, m] = hhmm.split(':').map(Number)
         const d = new Date(`${isoDate}T${String(h+4).padStart(2,'0')}:${String(m).padStart(2,'0')}:00.000Z`)
         return Math.floor(d.getTime() / 1000)
       }
       ```

       ```ts
       export default async function globalTeardown() {
       ```
