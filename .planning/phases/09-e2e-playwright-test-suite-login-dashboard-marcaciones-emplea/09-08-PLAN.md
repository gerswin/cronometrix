    ```ts
    import { test, expect } from '@playwright/test'
    import * as fs from 'node:fs/promises'
    import * as path from 'node:path'
    import { resetMutableTables, pushHikvisionEvent } from './fixtures/api'

    test.use({ storageState: 'e2e/.auth/admin.json' })

    test.describe('Dashboard (D-02 UAT depth)', () => {
      test.beforeEach(async ({ request }) => {
        await resetMutableTables(request)
      })

      test('renders 4 KPI tiles with Spanish labels', async ({ page }) => {
        await page.goto('/dashboard')
        await expect(page.getByTestId('kpi-empleados-presentes')).toBeVisible()
        await expect(page.getByTestId('kpi-retraso-hoy')).toBeVisible()
        await expect(page.getByTestId('kpi-dispositivos-activos')).toBeVisible()
        await expect(page.getByTestId('kpi-alertas-diurnas')).toBeVisible()
        // Spanish copy assertions (load-bearing per D-19)
        await expect(page.getByText('Empleados Presentes')).toBeVisible()
        await expect(page.getByText(/%\\s*Retraso\\s*Hoy/)).toBeVisible()
      })

      test('Empleados Presentes counts after pushing entry events', async ({ page, request }) => {
        // Push 2 entry events for EMP001 + EMP002 (seeded by Plan 03)
        const ana = await fs.readFile(
          path.resolve('./e2e/fixtures/hikvision-events/ana-entrada.xml'), 'utf8')
        const luis = await fs.readFile(
          path.resolve('./e2e/fixtures/hikvision-events/luis-entrada.xml'), 'utf8')
        await pushHikvisionEvent(request, ana)
        await pushHikvisionEvent(request, luis)

        await page.goto('/dashboard')
        // After backend processes events from the alertStream queue, KPI updates.
        // Use a generous timeout because the alertStream consumer may take 2-5s to fetch.
        await expect(page.getByTestId('kpi-empleados-presentes'))
          .toContainText(/[1-9]/, { timeout: 15_000 })
      })

      test('Distribución por Depto donut renders', async ({ page }) => {
        await page.goto('/dashboard')
        await expect(page.getByTestId('donut-by-dept')).toBeVisible()
        await expect(page.getByText(/Distribución por Depto/i)).toBeVisible()
      })

      test('ring buffer caps at 20 most-recent events', async ({ page, request }) => {
        // Push 25 events, assert at most 20 ring rows
        for (let i = 0; i < 25; i++) {
          const xml = await fs.readFile(
            path.resolve('./e2e/fixtures/hikvision-events/ana-entrada.xml'), 'utf8')
          // Mutate dateTime so each event is unique in time
          const mutated = xml.replace(
            'dateTime>2026-04-15T08:05:00-04:00',
            `dateTime>2026-04-15T08:0${(i % 10)}:0${(i % 10)}-04:00`
          )
          await pushHikvisionEvent(request, mutated)
        }
        await page.goto('/dashboard')
        const rows = page.getByTestId('ring-buffer').locator('[data-testid^="ring-row-"]')
        await expect(rows).toHaveCount(20, { timeout: 20_000 })
      })

      test('photo fallback to initials when image fails', async ({ page }) => {
        await page.goto('/dashboard')
        // Without a captured photo, every ring-row should show photo-fallback.
        // Push one event without any photo (current XML fixtures don't include JPEG part).
        const xml = await fs.readFile(
          path.resolve('./e2e/fixtures/hikvision-events/ana-entrada.xml'), 'utf8')
        await pushHikvisionEvent(page.request, xml)
        await page.reload()
        await expect(page.getByTestId('photo-fallback').first()).toBeVisible({ timeout: 15_000 })
      })

      test('SSE disconnect banner appears when backend kills the stream (eventual consistency)', async ({ page, request }) => {
        await page.goto('/dashboard')
        // Initially banner is hidden.
        await expect(page.getByTestId('sse-disconnect-banner')).toBeHidden()
        // Trigger disconnect: rely on a brief network throttling OR call __test_reset
        // (D-12 reset doesn't kill the SSE channel — instead, navigate away + back to
        // simulate a brief disconnect via context navigation).
        // Pragmatic approach: just verify the banner element EXISTS in DOM (hidden by default
        // class), so when SSE disconnects in real disconnect scenarios, the banner CAN show.
        // A true disconnect simulation requires backend control we don't have without modifying
        // the SSE route. Per RESEARCH §Pitfall 5, use eventual-consistency assertions only.
        // For Phase 9, we assert the banner element is present in DOM and has no flake-prone
        // backoff-timing assertion. Future plans can add a `?disconnect_sse=1` query param
        // to the backend's /events/stream route for controlled testing.
        const banner = page.getByTestId('sse-disconnect-banner')
        // Assert it exists in DOM (visible or not — confirms wiring); RESEARCH allows this.
        await expect(banner).toBeAttached()
      })

      test('empty state: no events yet → ring buffer renders empty list', async ({ page, request }) => {
        await resetMutableTables(request)
        await page.goto('/dashboard')
        const rows = page.getByTestId('ring-buffer').locator('[data-testid^="ring-row-"]')
        await expect(rows).toHaveCount(0)
        // Phase 4 D-6 may or may not show explicit empty copy; assert the container is empty.
        await expect(page.getByTestId('ring-buffer')).toBeVisible()
      })
    })
    ```

    NOTE: SSE banner test is intentionally weak (asserts attachment, not visibility transition) per RESEARCH §Pitfall 5 ("don't assert exact delays; trigger disconnect, assert eventual visibility"). Without a backend test-mode endpoint to kill the SSE stream, the strongest non-flaky test is "the element exists in DOM." Document this in the SUMMARY as a known limitation.
