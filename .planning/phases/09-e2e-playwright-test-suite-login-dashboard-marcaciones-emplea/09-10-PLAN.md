        const anaExit = await fs.readFile(path.resolve('./e2e/fixtures/hikvision-events/ana-salida.xml'), 'utf8')
        const luis = await fs.readFile(path.resolve('./e2e/fixtures/hikvision-events/luis-entrada.xml'), 'utf8')
        await pushHikvisionEvent(request, ana)
        await pushHikvisionEvent(request, anaExit)
        await pushHikvisionEvent(request, luis)
      })

      test('renders Reportes page with period selector', async ({ page }) => {
        await page.goto('/reports')
        await expect(page.getByText('Reportes')).toBeVisible()
        await expect(page.getByLabel(/Período|Periodo/i)).toBeVisible()

      test('weekly report: download Excel and verify content', async ({ page }) => {
        await page.goto('/reports')
        await page.getByLabel(/Período|Periodo/i).selectOption({ label: /Semanal/i })
        const dlPromise = page.waitForEvent('download')
        await page.getByRole('button', { name: /Excel|Exportar.*Excel/i }).click()
        const dl = await dlPromise
        const tmp = await dl.path()
        expect(tmp).toBeTruthy()
        const wb = XLSX.readFile(tmp!)
        const sheet = wb.Sheets[wb.SheetNames[0]]
        const rows = XLSX.utils.sheet_to_json<Record<string, unknown>>(sheet)
        expect(rows.length).toBeGreaterThan(0)
        // At least one row mentions Ana Pérez (seeded employee)
        const json = JSON.stringify(rows)
        expect(json).toMatch(/Ana Pérez/i)
      })

        await page.goto('/reports')
        await page.getByLabel(/Período|Periodo/i).selectOption({ label: /Mensual/i })

        await page.getByRole('button', { name: /Excel|Exportar.*Excel/i }).click()
